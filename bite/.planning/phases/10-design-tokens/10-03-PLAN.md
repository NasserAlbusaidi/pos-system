#   Spacing utilities: p-N, px-N, py-N, pt-N, pr-N, pb-N, pl-N, m-N, mx-N, my-N, mt-N, mr-N, mb-N, ml-N

    **Step 1 — Run the audit grep.**

    From the project root, run a per-file utility-class density count. Capture the output to a temp variable, then format as Markdown.

    ```bash
    # Count utility classes per blade file. Pattern matches the most common Tailwind
    # utility groups: spacing (p-/m-/gap-/space-), layout (flex/grid), typography
    # (text-/font-/leading-/tracking-), sizing (w-/h-).
    UTIL_REGEX='\b(p[xytrbl]?-[0-9.]+|m[xytrbl]?-[0-9.]+|gap(-[xy])?-[0-9.]+|space-[xy]-[0-9.]+|text-(xs|sm|base|lg|xl|[2-9]xl)|font-(thin|light|normal|medium|semibold|bold|extrabold|black)|leading-[a-z0-9]+|tracking-[a-z]+|flex|grid|inline-flex|inline-grid|w-[0-9./]+|h-[0-9./]+)\b'

    # Per-file count, sorted descending:
    for f in $(find resources/views -name '*.blade.php' | sort); do
      c=$(grep -oE "$UTIL_REGEX" "$f" 2>/dev/null | wc -l | tr -d ' ')
      if [ "$c" -gt 0 ]; then echo "$c $f"; fi
    done | sort -rn > /tmp/tailwind-audit.txt

    # Per-utility-token frequency (raw count + token, space-separated):
    grep -rohE "$UTIL_REGEX" resources/views | sort | uniq -c | sort -rn | head -50 > /tmp/tailwind-tokens.txt

    # Convert /tmp/tailwind-tokens.txt to a proper Markdown table:
    #   Input lines look like:   `  123 p-4`
    #   Output rows look like:   `| 123 | p-4 |`
    # The printf provides the table header and alignment row; awk emits each row.
    { printf '| Count | Utility |\n|---|---|\n'; awk '{print "| " $1 " | " $2 " |"}' /tmp/tailwind-tokens.txt; } > /tmp/tailwind-tokens-table.md
    ```
