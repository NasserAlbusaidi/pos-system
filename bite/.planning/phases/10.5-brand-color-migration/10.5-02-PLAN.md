<?php

namespace Tests\Feature;

use App\Models\Category;
use App\Models\Product;
use App\Models\Shop;
use App\Models\User;
use Illuminate\Foundation\Testing\RefreshDatabase;
use Tests\TestCase;

class BrandingCascadeTest extends TestCase
{
    use RefreshDatabase;

    /**
     * D-15 Test 1 — tenant route emits per-shop `<style>` override.
     * Locks the multi-tenant white-label promise in `app.blade.php` lines 27–89.
     */
    public function test_tenant_route_emits_shop_branding_style_block(): void
    {
        $shop = Shop::factory()->create([
            'name' => 'Sourdough Oman',
            'slug' => 'sourdough-test',
            'branding' => [
                'paper' => '#FFE4D6',
                'ink' => '#2C1810',
                'accent' => '#FF7A45',
            ],
        ]);

        $category = Category::create([
            'shop_id' => $shop->id,
            'name_en' => 'Bread',
            'name_ar' => 'خبز',
        ]);

        Product::forceCreate([
            'shop_id' => $shop->id,
            'category_id' => $category->id,
            'name_en' => 'Sourdough Loaf',
            'name_ar' => 'رغيف خميرة',
            'price' => 2.500,
            'is_available' => true,
            'is_visible' => true,
        ]);

        $response = $this->get(route('guest.menu', $shop->slug));

        $response->assertStatus(200);
        $response->assertSee('--paper: 255 228 214', false);    // #FFE4D6
        $response->assertSee('--ink: 44 24 16', false);         // #2C1810
        $response->assertSee('--crema: 255 122 69', false);     // #FF7A45
        $response->assertSee('--canvas:', false);               // derived neutral present
        $response->assertSee('--panel:', false);
        $response->assertSee('--ink-soft:', false);
    }

    /**
     * D-15 Test 2 — super-admin route uses platform default paper, never leaks tenant branding.
     * Custom-branded shop exists in DB but does not bleed into super-admin chrome.
     */
    public function test_super_admin_route_uses_platform_default_paper(): void
    {
        // Custom-branded shop in DB (would-be leak source)
        Shop::factory()->create([
            'name' => 'Sourdough Oman',
            'slug' => 'sourdough-test',
            'branding' => [
                'paper' => '#FFE4D6',
                'ink' => '#2C1810',
                'accent' => '#FF7A45',
            ],
        ]);

        $admin = User::factory()->superAdmin()->create();

        $response = $this->actingAs($admin)->get(route('super-admin.shops.index'));

        $response->assertStatus(200);
        // Sourdough's triple MUST NOT appear in super-admin HTML
        $response->assertDontSee('--paper: 255 228 214', false);
        $response->assertDontSee('--ink: 44 24 16', false);
        $response->assertDontSee('--crema: 255 122 69', false);
    }

    /**
     * D-15 Test 3 — admin /dashboard does not inject shop-branding.
     * Locks the D-11 strip: admin chrome is platform-only, no tenant cascade.
     */
    public function test_admin_dashboard_does_not_inject_shop_branding(): void
    {
        $shop = Shop::factory()->create([
            'name' => 'Sourdough Oman',
            'slug' => 'sourdough-test',
            'branding' => [
                'paper' => '#FFE4D6',
                'ink' => '#2C1810',
                'accent' => '#FF7A45',
            ],
        ]);

        $user = User::factory()->create([
            'shop_id' => $shop->id,
            'role' => 'admin',
        ]);

        $response = $this->actingAs($user)->get('/dashboard');

        $response->assertStatus(200);
        // Sourdough's custom triples MUST NOT appear — admin.blade.php no longer injects them
        $response->assertDontSee('--paper: 255 228 214', false);
        $response->assertDontSee('--ink: 44 24 16', false);
        $response->assertDontSee('--crema: 255 122 69', false);
    }
}
